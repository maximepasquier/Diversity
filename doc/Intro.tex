\subsection{Equations}

\begin{equation} \label{eq:continuity}
 \frac{\partial \rho}{\partial t}
 + \frac{\partial \rho u}{\partial t}
 + \frac{\partial \rho v}{\partial t}
 + \frac{\partial \rho w}{\partial t}
 = 0
\end{equation}
